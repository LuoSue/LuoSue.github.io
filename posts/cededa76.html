    parent[child] = find_root(parent[child], parent);

    if (root1 != root2) {

        if (child[root1] > child[root2]) {

            child[root1] += child[root2] + 1;
        }
        else {
            parent[root1] = root2;
            child[root2] += child[root1] + 1;
        }
        return true;
    }
    return false;
}
