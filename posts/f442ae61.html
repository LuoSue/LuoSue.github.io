 * };

class Solution {

    ListNode* fast = head;
    ListNode* slow = head;

        slow = slow->next;

    bool isPalindrome(ListNode* head) {
        ListNode* fast=head;
        ListNode* slow=head;
        while(fast&&fast->next){
            fast=fast->next->next;
            slow=slow->next;
        }
        ListNode* temp;
        ListNode* pre=NULL;
        while(slow!=NULL){
            temp=slow->next;
            slow->next=pre;
            pre=slow;
            slow=temp;
        }
        while(pre!=NULL){
            if((head->val)!=(pre->val)){return false;}
            head=head->next;pre=pre->next;  
        }
        return true;
    }
};
